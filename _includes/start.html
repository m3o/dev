# Install the micro binary
curl -fsSL https://install.m3o.com/micro | /bin/bash

# Set the environment to the platform
micro env set platform

# Signup before getting started
micro signup

# Login with the credentials provided during signup.
micro login

# Create a new service (follow the instructions and push to Github)
micro new helloworld

# Deploy the service (this is a public sample)
micro run github.com/micro/services/helloworld

# Check the service status
micro status

# Query the logs
micro logs helloworld

# Call the service
micro helloworld call --name=Alice

# Deploy new changes
micro update helloworld

# Get your namespace
NAMESPACE=$(micro user namespace)

# Call service via the public http API
curl "https://$NAMESPACE.m3o.app/helloworld?name=Alice"
